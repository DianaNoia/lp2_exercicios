## Problema

2 - Escreve um programa que realize a multiplicação entre uma matriz _n x m_
e um vetor _m x 1_. O programa deve começar por solicitar as dimensões _n_ e
_m_, pedindo depois ao utilizador que preencha os valores da matriz e do vetor
a multiplicar, e finalmente mostrando o resultado final na forma de um vetor
_n x 1_.

_Nota 1:_ Este exercício deve ser resolvido com recurso a _arrays_
multidimensionais.

_Nota 2:_ A resolução deste exercício depende da matéria de
[multiplicação de matrizes](https://mathinsight.org/matrix_vector_multiplication)
(disciplina de Introdução à Matemática e Física para Jogos I).

_Nota 3:_ Podes experimentar multiplicar uma matriz por um vetor _online_ em
http://matrix.reshish.com/multiplication.php.

```cs
using System;

namespace Ex2
{
    class Program
    {
        static void Main(string[] args)
        {
            // Declaras as variaveis necessarias
            int y, x;
            float[,] matrix;
            float[] vector;
            float[] multiply;

            // Pede o input ao utilizador
            Console.WriteLine("Introduz o numero de colunas da matriz:");
            x = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Introduz o numero de linhas da matriz:");
            y = Convert.ToInt32(Console.ReadLine());

            // Gera a Matriz
            matrix = new float[y, x];

            // Gera array que recebe a multiplicação
            multiply = new float[y];

            // Gera o Vetor
            vector = new float[x];

            // Percorre o array para ser preenchido
            for (int i = 0; i < y; i++)
            {
                for (int j = 0; j < x; j++)
                {
                    Console.Clear();
                    Console.WriteLine("Preenche a matriz:");
                    Console.WriteLine($"x:{j}, y:{i}");
                    matrix[i, j] = Convert.ToSingle(Console.ReadLine());
                }
            }

            // Percorre o array para ser preenchido
            for (int i = 0; i < x; i++)
            {
                Console.Clear();
                Console.WriteLine("Preenche o vetor:");
                Console.WriteLine($"Casa nº:{i}");
                vector[i] = Convert.ToSingle(Console.ReadLine());
            }

            // Multiplica a Matriz com o Vetor
            for (int i = 0; i < y; i++)
            {
                float mp = 0;
                for (int j = 0; j < x; j++)
                {
                    mp += matrix[i, j] * vector[j];
                }
                multiply[i] = mp;
            }

            // Mostra o Resultado
            Console.Write($"Resultado: ");
            for (int i = 0; i < multiply.Length; i++)
            {
                Console.Write(multiply[i] + ", ");
            }
        }
    }
}
```

```csharp
namespace Exercise02
{
    public class MatrixCalculator
    {
        private readonly AppUI appUI;

        public Matrix MatrixA { get; private set; }
        public Matrix MatrixB { get; private set; }
        public Matrix Result => MatrixA * MatrixB;

        public MatrixCalculator()
        {
            appUI = new AppUI(this);
        }

        public void Run()
        {
            bool status = true;

            while (status)
            {
                if (MatrixA == null)
                {
                    MatrixA = new Matrix(appUI.PromptMatrixSize(), "Matrix", 'M');

                    // Exercise asks Matrix * Vector so the 1 column is forced
                    // Calculator supports any value
                    MatrixB = new Matrix((MatrixA.Size.columns, 1), "Vector", 'V');

                    appUI.PromptMatrixValues(MatrixA);

                    appUI.PromptMatrixValues(MatrixB);
                }

                status = appUI.Refresh();
            }
        }

        public void Reset()
        {
            MatrixA = null;

            MatrixB = null;
        }
    }
}
```

Class Matrix:

```csharp
using System;

namespace Exercise02
{
    public class Matrix
    {
        private int rows, columns;

        public string Name { get; }
        public char Identifier { get; }
        public float?[,] Values { get; private set; }
        public (int rows, int columns) Size => (rows, columns);
        public bool HasAllValuesSet
        {
            get
            {
                for (int i = 0; i < Size.rows; i++)
                {
                    for (int k = 0; k < Size.columns; k++)
                    {
                        if (!Values[i, k].HasValue)
                            return false;
                    }
                }

                return true;
            }
        }

        public Matrix((int rows, int columns) dimensions, string name, char id)
        {
            this.rows = dimensions.rows;
            this.columns = dimensions.columns;

            Values = new float?[rows, columns];

            Name = name;

            Identifier = id;
        }

        // Sets value at given index.
        public bool SetValue((int row, int column, float value) data)
        {
            // Checks if index is inside of array
            if (data.row - 1 >= Size.rows || data.column - 1 >= Size.columns)
            {
                return false;
            }

            Values[data.row - 1, data.column - 1] = data.value;

            return true;
        }

        // Multiplication operator overload
        public static Matrix operator *(Matrix a, Matrix b)
        {
            Matrix result = new Matrix((a.Size.rows, b.Size.columns), "Result", 'R');

            if (a.Size.columns != b.Size.rows)
            {
                throw new ArgumentException("2nd Matrix.Size.rows == 1st Matrix.Size.columns");
            }

            for (int i = 0; i < result.Size.rows; i++)
            {
                for (int j = 0; j < b.Size.columns; j++)
                {
                    float mult = 0;

                    for (int k = 0; k < a.Size.columns; k++)
                    {
                        float aValue = a.Values[i, k].Value;
                        float bValue = b.Values[k, j].Value;

                        mult += aValue * bValue;
                    }

                    result.Values[i, j] = mult;
                }
            }

            return result;
        }
    }
}
```

Class AppUI:

```csharp
using System;

namespace Exercise02
{
    public class AppUI
    {
        private readonly MatrixCalculator calculator;

        public AppUI(MatrixCalculator calculator)
        {
            this.calculator = calculator;
        }

        // Asks user of matrix size
        public (int rows, int columns) PromptMatrixSize()
        {
            string[] input;
            string errorMsg = string.Empty;
            (int rows, int columns) result = (0, 0);

            while (result.rows < 1 || result.columns < 1)
            {
                Console.Clear();

                if (!string.IsNullOrEmpty(errorMsg))
                    Console.WriteLine(errorMsg);

                Console.Write("Set initial matrix dimensions (type \"NxM\"): ");

                input = Console.ReadLine().Split('x');

                // Input is valid?
                if (input.Length < 2 || !int.TryParse(input[0], out result.rows)
                    || !int.TryParse(input[1], out result.columns))
                {
                    errorMsg = "Format must be \"NxM\". N is rows, M is columns.";
                }
            }

            return result;
        }

        // Recursive method to set all values of matrix
        public void PromptMatrixValues(Matrix matrix)
        {
            while (!matrix.HasAllValuesSet)
            {
                Console.Clear();

                PrintMatrix(matrix);

                Console.CursorTop += 1;

                for (int i = 0; i < matrix.Size.rows; i++)
                {
                    for (int k = 0; k < matrix.Size.columns; k++)
                    {
                        if (!matrix.Values[i, k].HasValue)
                        {
                            Console.Write($"Set {matrix.Identifier}{i + 1}{k + 1}: ");

                            Helpers.TryParse(Console.ReadLine(), out matrix.Values[i, k]);

                            PromptMatrixValues(matrix);
                        }
                    }
                }
            }
        }

        public bool Refresh()
        {
            Console.Clear();

            PrintMatrix(calculator.MatrixA);

            Console.CursorTop += 1;

            PrintMatrix(calculator.MatrixB);

            Console.CursorTop += 1;

            PrintMatrix(calculator.Result);

            Console.CursorTop += 1;

            return PromptConsole();
        }

        private void PrintMatrix(Matrix matrix)
        {
            Console.WriteLine($"{matrix.Name} Values:");

            for (int i = 0; i < matrix.Size.rows; i++)
            {
                for (int k = 0; k < matrix.Size.columns; k++)
                {
                    if (matrix.Values[i, k].HasValue)
                        Console.Write($"\t{matrix.Values[i, k].Value}\t");
                    else
                        Console.Write($"\t{matrix.Identifier}{i + 1}{k + 1}\t");
                }

                Console.CursorLeft = 0;
                Console.CursorTop += 1;
            }
        }

        private bool PromptConsole()
        {
            string input;

            Console.WriteLine("Available commands:");

            Console.WriteLine("\t set m <row> <column> <value> : sets matrix value.");

            Console.WriteLine("\t set v <row> <column> <value> : sets vector value");

            Console.WriteLine("\t reset : allows to set a new size for the matrix.");

            Console.WriteLine("\t quit/exit : quits calculator.");

            while (true)
            {
                Console.CursorTop += 1;

                Console.Write("> ");

                input = Console.ReadLine().ToLower();

                if (input == "quit" || input == "exit")
                {
                    return false;
                }
                else if (input == "reset")
                {
                    calculator.Reset();

                    return true;
                }
                else if (input.StartsWith("set m"))
                {
                    if (TryParseSetInput(input,
                            out (int row, int column, float value) parsed))
                    {
                        if (calculator.MatrixA.SetValue(parsed))
                            return true;
                        else
                            Console.WriteLine("Failed to set value. Check syntax.");
                    }
                    else
                    {
                        Console.WriteLine("Invalid syntax.");
                    }
                }
                else if (input.StartsWith("set v"))
                {
                    if (TryParseSetInput(input,
                            out (int row, int column, float value) parsed))
                    {
                        if (calculator.MatrixB.SetValue(parsed))
                            return true;
                        else
                            Console.WriteLine("Failed to set value. Check syntax.");
                    }
                    else
                    {
                        Console.WriteLine("Invalid syntax.");
                    }
                }
                else
                {
                    Console.WriteLine("Unknown command...");
                }
            }
        }

        // TryParse for set v/m <row> <column> <value> command
        private bool TryParseSetInput(string inInput,
            out (int row, int column, float value) parsed)
        {
            string[] input = inInput.Split();

            parsed.row = 0;
            parsed.column = 0;
            parsed.value = 0;

            if (input.Length != 5)
                return false;

            if (int.TryParse(input[2], out parsed.row))
                if (int.TryParse(input[3], out parsed.column))
                    if (float.TryParse(input[4], out parsed.value))
                        return true;

            return false;
        }
    }
}
```

Class Helpers:

```csharp
namespace Exercise02
{
    public static class Helpers
    {
        // Tries to parse nullable float
        public static bool TryParse(string s, out float? result)
        {
            result = null;

            if (float.TryParse(s, out float x))
            {
                result = x;
            }

            return result.HasValue;
        }
    }
}
```

```csharp
using System;

namespace _01_02
{
    class Program
    {
        // Matriz
        private static float[,] Matrix { get; set; }

        // Vetor
        private static float[,] Vector { get; set; }

        // Resultado final 
        private static float[,] Result { get; set; }

        // Número de colunas da matriz
        private static int matrixColumns { get; set; }

        // Número de linhas da matriz
        private static int matrixLines { get; set; }


        private static void Main(string[] args) {
            // Método para criar a matriz
            ConstructMatrix();

            // Método para criar o vetor
            ConstructVector();

            // Método para multiplicar a matriz com o vetor
            Multiply();

            // Método para somar a matriz
            Sum();

            // Método para expor o resultado
            PrintResult();
        }

        // Método para criar a matriz
        private static void ConstructMatrix() {
            // Input do user para definir o número de linhas
            string inputMatrixLines;

            // Input do user para definir o número de colunas da matriz
            string inputMatrixColumns;

            // Verificar se ficou definido o número de linhas da matriz.
            bool isMatrixLines = false;

            // Verificar se ficou definido o número de colunas da matriz.
            bool isMatrixColumns = false;

            // Número de linhas da matriz.
            int matrixLineValue;

            // Número de colunas da matriz
            int matrixColumnValue;

            // Input do user
            string userInputValue;

            // Valor dos elementos
            float value = 0f;

            // Verificar se os elementos estão preenchidos
            bool isValue = false;

            // Verificar se foi colocado um elemento.
            bool isAdded = false;


            // Definir o número de linhas
            while (isMatrixLines == false) {
                Console.WriteLine("Qual é o número de linhas da matriz?");

                // User Input
                inputMatrixLines = Console.ReadLine();

                // Tentar convertar o input para int
                if (Int32.TryParse(inputMatrixLines, out matrixLineValue)) {
                    matrixLines = matrixLineValue;
                    isMatrixLines = true;
                } else {
                    Console.WriteLine("Valor inválido, tente de novo.");
                }
            }
            

            // Definir o número de colunas
            while (isMatrixColumns == false) {
                Console.WriteLine("Qual é o número de colunas da matriz?");

                // User input
                inputMatrixColumns = Console.ReadLine();

                // Tentar converter o input para int
                if (Int32.TryParse(inputMatrixColumns,out matrixColumnValue)) {
                    matrixColumns = matrixColumnValue;
                    isMatrixColumns = true;
                } else {
                    Console.WriteLine("Valor inválido, tente de novo.");
                }
            }

            // Colocar o número de linhas e colunas na matriz
            Matrix = new float[matrixLines, matrixColumns];
        
            Console.WriteLine($"A matriz tem {Matrix.Length} elementos");

            // Definir os valores
            while (isValue == false) {
                for (int j = 0; j < matrixLines; j++) {
                    for (int i = 0; i < matrixColumns; i++) {
                        while (isAdded == false) {
                            Console.WriteLine($"Insere um número na linha" +
                                $" {j + 1} na coluna {i + 1}: ");
                            if (j == 0 && i == 0) {
                                // Mostrar matriz 
                                PrintMatrix();
                            }
                            // User input do valor
                            userInputValue = Console.ReadLine();

                            // Tentar converter o input para float
                            if (float.TryParse(userInputValue, out value)) {
                                // Colocar na linha
                                Matrix[j, i] = value;

                                // Mostrar matriz
                                PrintMatrix();

                                isAdded = true;
                            } else {
                                Console.WriteLine("Valor inválido, " +
                                    "tente de novo");
                            }
                        }
                        isAdded = false;
                    }
                    isValue = true;
                }
            }
        }

        // Método para criar o vetor
        private static void ConstructVector() {
            // Criar o vetor
            Vector = new float[matrixColumns, 1];
            Console.WriteLine($"O vector tem {Vector.Length} elementos");

            // Definir os valores
            // Input do user
            string userInput;

            // Valor
            float value = 0f;

            // Verificar se foi inserido algum valor
            bool isAdded = false;
 
            for (int i = 0; i < matrixColumns; i++) {
                while (isAdded == false) {
                    Console.WriteLine($"Insere um número na linha " +
                        $"{i + 1} na coluna 1: ");
                    if (i == 0) {
                        // Mostrar vector
                        PrintVector();
                        
                    }
                    // Input do user do valor
                    userInput = Console.ReadLine();

                    // Converter o input para int
                    if (float.TryParse(userInput, out value)) {
                        value = float.Parse(userInput);
                        // Colocar na linha
                        Vector[i, 0] = value;
                        
                        // Mostrar vector
                        PrintVector();
                        
                        isAdded = true;
                    } else {
                        Console.WriteLine("Valor inválido, tente de novo");
                    }
                }
                isAdded = false;
            }
        }

        // Método para multiplicar a matriz com o vetor
        private static void Multiply() {
            Console.WriteLine("Calculate...");

            // Index do vetor
            int vectorIndex = 0;

            // Multiplicar os valores da matriz com o vetor
            for (int n = 0; n < Matrix.GetLength(0); n++) {
                for (int i = 0; i < Matrix.GetLength(1); i++) {
                    while (i < Vector.GetLength(0)) {
                        Matrix[n, i] = Matrix[n, i]
                                * Vector[vectorIndex, 0];
                        vectorIndex++;
                        i++;
                    }
                }
                vectorIndex = default;
            }
        }

        // Método para somar os valores da matriz
        private static void Sum() {
            // Definir o tamanho do result
            Result = new float[Matrix.GetLength(0), 1];

            // Guardar soma para meter no result
            float sum;

            // Somar os valores da matriz
            for (int j = 0; j < Matrix.GetLength(0); j++) {
                sum = default;
                for (int i = 0; i < Matrix.GetLength(1); i++) {
                    sum = sum + Matrix[j, i];
                }
                // Colocar na linha
                Result[j, 0] = sum;
            }
        }

        // Método para mostrar o resultado 
        private static void PrintResult() {
            // Tamanho da linha
            int rowLength = Result.GetLength(0);
            
            // Tamanho da coluna
            int colLength = Result.GetLength(1);
            
            // Resultado
            string resultString = "";
            
            for (int i = 0; i < rowLength; i++) {
                for (int j = 0; j < colLength; j++) {
                    // Colocar os valores de result
                    resultString += Result[i, j].ToString();
                }
                // Colocar nova linha 
                resultString += Environment.NewLine + Environment.NewLine;
            }
            // Mostrar resultado
            Console.WriteLine("Resultado:");
            Console.WriteLine(resultString);
            
        }

        // Método para mostrar a matriz 
        private static void PrintMatrix() {
            // Tamanho da linha
            int rowLength = Matrix.GetLength(0);
            
            // Tamanho da coluna
            int colLength = Matrix.GetLength(1);
            
            // Resultado
            string matrixString = "";
            

            for (int i = 0; i < rowLength; i++) {
                for (int j = 0; j < colLength; j++) {
                    // Colocar os valores de result
                    matrixString += Matrix[i, j].ToString() + " ";           
                }
                // Colocar nova linha 
                matrixString += Environment.NewLine + Environment.NewLine;
                
            }
            // Mostrar resultado
            Console.WriteLine("Matriz: ");
            Console.WriteLine(matrixString);          
        }

        // Método para mostrar o vector 
        private static void PrintVector() {
            // Tamanho da linha
            int rowLength = Vector.GetLength(0);
            
            // Tamanho da coluna
            int colLength = Vector.GetLength(1);
            
            // Resultado
            string matrixString = "";
            

            for (int i = 0; i < rowLength; i++) {
                for (int j = 0; j < colLength; j++) {
                    // Colocar os valores de result
                    matrixString += Vector[i, j].ToString() + " ";                   
                }
                // Colocar nova linha 
                matrixString += Environment.NewLine + Environment.NewLine;                
            }
            // Mostrar resultado
            Console.WriteLine("Vector: ");
            Console.WriteLine(matrixString);
        }
    }
}
```
